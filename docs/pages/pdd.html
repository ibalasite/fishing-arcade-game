§1 設計原則與視覺方向

§1.1 整體風格定義

風格定位：海洋電玩街機（Ocean Neon Arcade）

融合深海水族館的沉浸氛圍與亞洲電玩街機的霓虹張力，目標是讓玩家啟動 App 的第一眼就喚起「站在街機廳前、投幣準備開戰」的記憶感，同時兼具手機端的精緻品質。

四個核心設計關鍵字：

  1. 深海沉浸（Deep Ocean Immersion）：深邃藍黑的海底背景，光線從水面折射而下，魚群在玻璃質感的水中層次游動。
  2. 霓虹爽感（Neon Punch）：砲台射擊、命中特效、Jackpot 引爆均使用飽和霓虹色系，形成強烈的視覺快感。
  3. 即時競技張力（Real-Time Arena）：HUD 設計強調「誰在做什麼」的即時感知，多人存在感顯而易見。
  4. 亞洲街機質感（Asian Arcade Heritage）：金色光效、繁體中文粗體、倍率數字字體選用等細節致敬實體街機廳視覺語言。

風格方向參照（Style Direction）：Neo-brutalism × Glassmorphism × Arcade Neon，選擇這三者的混合而非任一單一方向，理由是：

  • Neo-brutalism 帶來倍率數字的粗壯視覺衝擊
  • Glassmorphism 實現水下深度感（半透明 HUD 覆蓋海底場景）
  • Arcade Neon 提供核心情感張力（霓虹光暈、爆炸特效）

§1.3 字體系統

遊戲標題字體（Logo / Title）

  • 字體：Noto Sans TC ExtraBold（900 weight）+ 霓虹光效後處理
  • 場景：App 啟動畫面、主畫面遊戲名稱
  • 字級：clamp(2.5rem, 8vw, 5rem)（適配手機到平板）
  • 光效：text-shadow: 0 0 12px #00F5FF, 0 0 24px #00F5FF80（Cocos 實作：LabelComponent + 後處理）
  • 備援：系統字體不支援時降級為 Noto Sans TC Bold

UI 標籤字體（Button / HUD Labels）

  • 字體：Noto Sans TC Medium（500 weight）
  • 場景：按鈕文字、HUD 玩家名稱、房間 ID、選單項目
  • 字級：14sp–18sp（Cocos Creator sp 單位，自動適配螢幕密度）
  • 顏色規則：亮色場景 #FFFFFF，深色場景 #E0E8FF

數字顯示字體（Numbers / Currency）

  • 字體：Orbitron Bold（英數專用）或 Noto Sans TC Bold（含中文數字）
  • 場景：金幣餘額、鑽石數量、Jackpot 累積金額、倍率顯示、砲台冷卻倒數
  • 字級：金幣/鑽石 20sp，Jackpot 主數字 clamp(28sp, 5vw, 48sp)，倍率小標 12sp
  • 特效：Jackpot 數字採滾動動畫（NumberRoller 組件，詳見 §3.3）；金幣增加時使用 +N 浮動文字動畫

說明文字字體（Body / Privacy / Legal）

  • 字體：Noto Sans TC Regular（400 weight）
  • 場景：隱私政策彈窗、帳號刪除說明、Jackpot 機率揭示、系統提示
  • 字級：12sp–14sp
  • 行距：1.6（符合 WCAG 可讀性建議）
  • 顏色：#C8D6E5（低對比正文）；必要法規說明升級為 #FFFFFF

§2.1.2 關鍵元素設計規格

§2.2.2 海底背景設計

背景層次（由後至前）：

  1. Layer 0 — 深海底部：漸層色，#050D1A → #0B2447（垂直漸層）
  2. Layer 1 — 珊瑚礁裝飾：靜態 Sprite，左下/右下角，低飽和度（不搶焦點）
  3. Layer 2 — 中景水草：緩慢擺動動畫（SkeletonAnimation，幅度小）
  4. Layer 3 — 光線射入效果：粒子系統（自上而下，白色半透明光柱，低不透明度 0.15）
  5. Layer 4 — 前景氣泡：粒子系統（隨機上升小氣泡，loop）

背景動畫規格：

  • 全部使用 compositor-friendly 屬性（transform / opacity）
  • 水草擺動週期：4-6s（隨機化，避免同步感）
  • 光線粒子：生命週期 8s，spawn rate 2/s，透明度 0→0.15→0

@ccclass('NumberRoller')
export class NumberRoller extends Component {
    @property(Label) displayLabel: Label = null;
    @property public duration: number = 0.3;  // 滾動動畫時長（秒）

    // F3 fix: CC4.x tween 只能動畫化 public 屬性，private `_currentValue` 會被忽略

    private _targetValue: number = 0;
    private _tween: Tween<this> = null;

        this._targetValue = newValue;

        // 使用 tween 動畫化 public currentValue，onUpdate 每幀同步 Label

                    // 格式化為帶千位分隔符的字串
                    this.displayLabel.string =
                        Math.floor(this.currentValue).toLocaleString('zh-TW');
                }
            })
            .start();
    }
}

§4.1 Safe Area 處理

§4.1.1 Cocos Creator 4.x 安全區域實作

typescript
// components/layout/SafeAreaAdapter.ts
@ccclass('SafeAreaAdapter')
export class SafeAreaAdapter extends Component {
    @property(Node) topInset: Node = null;
    @property(Node) bottomInset: Node = null;
    @property(Node) leftInset: Node = null;
    @property(Node) rightInset: Node = null;

    onLoad() {
        // 使用 Cocos Creator 4.x 的 sys.getSafeAreaRect() 取得安全區域
        // sys.getSafeAreaRect() 與 screen.windowSize 均以實體像素（physical pixels）為單位
        // UITransform.setContentSize() 接受邏輯單位（設計解析度 design resolution units）
        // → 必須除以 devicePixelRatio 轉換，否則在 3x DPI 裝置上 inset 會放大 3 倍
        const safeArea = sys.getSafeAreaRect();  // physical pixels
        const screenSize = screen.windowSize;    // physical pixels
        const dpr = screen.devicePixelRatio || 1;  // Round 5 F2 fix: 實體→邏輯像素轉換比例

        // Y 軸向上：safeArea.y = 底部實體像素 inset；頂部需反算
        const topPad    = (screenSize.height - safeArea.y - safeArea.height) / dpr;
        const bottomPad = safeArea.y / dpr;
        const leftPad   = safeArea.x / dpr;
        const rightPad  = (screenSize.width - safeArea.x - safeArea.width) / dpr;
        const logicalW  = screenSize.width  / dpr;
        const logicalH  = screenSize.height / dpr;

        // UITransform.setContentSize 接受邏輯單位
        this.topInset.getComponent(UITransform).setContentSize(logicalW, topPad);
        this.bottomInset.getComponent(UITransform).setContentSize(logicalW, bottomPad);
        this.leftInset.getComponent(UITransform).setContentSize(leftPad, logicalH);

§4.3.2 記憶體管理策略

ObjectPool 設計（避免 GC 峰值）：

typescript
// utils/ObjectPoolManager.ts
// F4 fix + F-R2-01/F-R2-02 fix + F2 fix (Round 3):
// CC4.x ObjectPool<T> 接受 creator + handler 物件；singleton 以 getInstance() 存取
export class ObjectPoolManager {
    private static _instance: ObjectPoolManager;

    /** F2 fix: 全域 singleton，跨場景存取必須用此方法 */
    public static getInstance(): ObjectPoolManager {
        if (!ObjectPoolManager._instance) {
            ObjectPoolManager._instance = new ObjectPoolManager();
        }
        return ObjectPoolManager._instance;
    }

    private _pools: Map<string, ObjectPool<Node>> = new Map();
    private _prefabs: Map<string, Prefab> = new Map();

    /** 註冊 Prefab，在 Boot.scene 或 GameRoom.scene 初始化時呼叫 */
    public registerPrefab(key: string, prefab: Prefab): void {
        this._prefabs.set(key, prefab);
    }

    public clearAllPools(): void {
        this._pools.forEach((pool) => pool.clear());
        this._pools.clear();
    }

    getPool(prefabKey: string, warmUpCount: number): ObjectPool<Node> {
        if (!this._pools.has(prefabKey)) {
            const prefab = this._prefabs.get(prefabKey);
            // CC4.x ObjectPool handler 介面：IPoolHandlerComponent<T> 定義為 { onFree, reuse }
            // （Round 5 F1 驗證：onFree = 回收時呼叫；reuse = 取出時呼叫，符合 CC4.x TypeScript 型別定義）
            const pool = new ObjectPool<Node>(
                () => instantiate(prefab),
                {
                    onFree: (node) => { node.active = false; },  // 回收入池時隱藏
                    reuse: (node) => { node.active = true; }     // 從池中取出時顯示
                }
            );
            // 預熱 N 個節點（避免遊戲中動態 instantiate 造成 GC 峰值）
            for (let i = 0; i < warmUpCount; i++) {
                pool.put(instantiate(prefab));
            }
            this._pools.set(prefabKey, pool);
        }
        return this._pools.get(prefabKey);
    }
}

accessibilityLabel 設定方式（F12 fix）：

Cocos Creator 4.x 沒有直接的 accessibilityLabel 屬性，需透過 jsb.reflection 呼叫原生 API。下列工具函式封裝此邏輯：

typescript
// utils/AccessibilityHelper.ts
// F-R2-04 fix: 移除 _uiTransformComp.nativeObject 偽造屬性查詢（始終為 undefined）
// 直接使用 node.uuid 作為原生端識別符（jsb.reflection 呼叫中已正確使用）
export function setAccessibilityLabel(node: Node, label: string, hint?: string): void {
    if (!jsb) return;  // Web 平台跳過
    if (!node) return;

    if (sys.os === sys.OS.IOS) {
        // iOS：AppController 原生端以 uuid 查找對應 UIView，設定 accessibilityLabel
        jsb.reflection.callStaticMethod(
            'AppController',
            'setAccessibilityLabel:hint:forNodeId:',
            `${label}|${hint ?? ''}|${node.uuid}`
        );
    } else if (sys.os === sys.OS.ANDROID) {
        // Android：Cocos2dxHelper 原生端以 uuid 查找對應 View，設定 contentDescription
        jsb.reflection.callStaticMethod(
            'org/cocos2dx/lib/Cocos2dxHelper',
            'setAccessibilityLabel',
            '(Ljava/lang/String;Ljava/lang/String;)V',
            node.uuid,
            label
        );
    }
}

typescript
// utils/MotionPreference.ts
export function isReduceMotionEnabled(): boolean {
    if (!jsb) return false;  // Web 平台預設不減少動態效果

    if (sys.os === sys.OS.IOS) {
        // F4 fix: UIAccessibilityIsReduceMotionEnabled 是 C 函式 / ObjC property，
        // 需透過在 AppController.mm 封裝的靜態 ObjC 方法呼叫（CC4.x jsb.reflection 限制）
        // AppController.mm 需加入：
        //   + (NSString *)isReduceMotionEnabled {
        //       return UIAccessibilityIsReduceMotionEnabled() ? @"true" : @"false";
        //   }
        return jsb.reflection.callStaticMethod(
            'AppController',
            'isReduceMotionEnabled',
            ''
        ) === 'true';
    }

    if (sys.os === sys.OS.ANDROID) {
        // F13 fix: Android 透過 Settings.Global.ANIMATOR_DURATION_SCALE 偵測
        // ANIMATOR_DURATION_SCALE = 0 時，使用者已在開發者選項中關閉動畫（等同 Reduce Motion）
        // 正常值為 1.0；無障礙設定的「移除動畫」也會設為 0
        // F-R2-14 fix: JNI '()F' 回傳值在 JSB 中為 JS number，不需 string 轉型
        const scale = jsb.reflection.callStaticMethod(
            'org/cocos2dx/lib/Cocos2dxHelper',
            'getAnimatorDurationScale',
            '()F'
        ) as number;
        return scale === 0;
    }

    return false;
}
